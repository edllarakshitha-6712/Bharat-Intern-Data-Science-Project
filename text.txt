import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
df = pd.read_csv('reviews_data.csv')

print("First 5 rows:")
print(df.head())

print("\nDataset Info:")
print(df.info())

print("\nMissing values:")
print(df.isnull().sum())

df['Date'] = df['Date'].str.replace('Reviewed ', '', regex=False)

df['Date'] = pd.to_datetime(df['Date'], format='%b. %d, %Y', errors='coerce')

df['Review'] = df['Review'].fillna('')



# Rating distribution
plt.figure()
df['Rating'].value_counts().sort_index().plot(kind='bar')
plt.title("Rating Distribution")
plt.xlabel("Rating")
plt.ylabel("Count")
plt.show()

# Reviews over time
plt.figure()
df.groupby(df['Date'].dt.year)['Review'].count().plot()
plt.title("Reviews per Year")
plt.xlabel("Year")
plt.ylabel("Number of Reviews")
plt.show()

import re
import nltk
nltk.download('stopwords')
from nltk.corpus import stopwords

stop_words = set(stopwords.words('english'))

def clean_text(text):
    text = text.lower()
    text = re.sub(r'[^a-z\s]', '', text)
    words = text.split()
    words = [word for word in words if word not in stop_words]
    return " ".join(words)

df['clean_review'] = df['Review'].apply(clean_text)

from textblob import TextBlob

def get_sentiment(text):
    return TextBlob(text).sentiment.polarity

df['sentiment'] = df['clean_review'].apply(get_sentiment)


df['sentiment_label'] = df['sentiment'].apply(
    lambda x: 'Positive' if x > 0 else ('Negative' if x < 0 else 'Neutral')
)

print("\nSentiment Counts:")
print(df['sentiment_label'].value_counts())

plt.figure()
df['sentiment_label'].value_counts().plot(kind='pie', autopct='%1.1f%%')
plt.title("Sentiment Distribution")
plt.ylabel('')
plt.show()


df.to_csv('processed_reviews.csv', index=False)

print("Processed file saved as processed_reviews.csv")
